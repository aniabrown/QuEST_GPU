158 void rotateQubit(MultiQubit multiQubit, const int rotQubit, Complex alpha, Complex beta)
159 
160 {
161  // flag to require memory exchange. 1: an entire block fits on one rank, 0: at most half a block fits on one rank
162  int useLocalDataOnly = halfMatrixBlockFitsInChunk(multiQubit.numAmps, rotQubit);
163  Complex rot1, rot2;
164 
165  // rank's chunk is in upper half of block
166  int rankIsUpper;
167  int pairRank; // rank of corresponding chunk
168 
169  // MPI send/receive vars
170  int TAG=100;
171  MPI_Status status;
172 
173  if (useLocalDataOnly){
174  // all values required to update state vector lie in this rank
175  rotateQubitLocal(multiQubit, rotQubit, alpha, beta);
176  } else {
177  // need to get corresponding chunk of state vector from other rank
178  rankIsUpper = chunkIsUpper(multiQubit.chunkId, multiQubit.numAmps, rotQubit);
179  getRotAngle(rankIsUpper, &rot1, &rot2, alpha, beta);
180  pairRank = getChunkPairId(rankIsUpper, multiQubit.chunkId, multiQubit.numAmps, rotQubit);
181  //printf("%d rank has pair rank: %d\n", multiQubit.rank, pairRank);
182  // get corresponding values from my pair
183  MPI_Sendrecv(multiQubit.stateVec.real, multiQubit.numAmps, MPI_DOUBLE, pairRank, TAG,
184  multiQubit.pairStateVec.real, multiQubit.numAmps, MPI_DOUBLE, pairRank, TAG,
185  MPI_COMM_WORLD, &status);
186  //printf("rank: %d err: %d\n", multiQubit.rank, err);
187  MPI_Sendrecv(multiQubit.stateVec.imag, multiQubit.numAmps, MPI_DOUBLE, pairRank, TAG,
188  multiQubit.pairStateVec.imag, multiQubit.numAmps, MPI_DOUBLE, pairRank, TAG,
189  MPI_COMM_WORLD, &status);
190  // this rank's values are either in the upper of lower half of the block. send values to rotateQubitDistributed
191  // in the correct order

193  rotateQubitDistributed(multiQubit,rotQubit,rot1,rot2,
194  multiQubit.stateVec, //upper

196  multiQubit.stateVec); //output
197  } else {
